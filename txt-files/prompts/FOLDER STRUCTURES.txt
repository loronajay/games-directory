games-directory-page/

│
├── games/                   # Individual arcade game runtimes
│   │
│   ├── apple-catcher/
│   ├── art-of-war/
│   ├── bird-duty/

│
├── invest/                  # Investor page content and materials
│
├── js/                      # Core platform JavaScript systems
│   │
│   ├── arcade-input.js      # Unified input system
│   └── jay-mobile.js        # Mobile controller engine
│
├── music/                   # Background music and audio assets
│
├── previews/                # Game preview videos for the grid
│
├── scripts/                 # Build pipeline and automation tools
│
├── txt-files/               # Development documentation and templates
│   │
│   ├── grid-page/           # Grid page structure templates
│   ├── index-page/          # Arcade OS index page templates
│   ├── js/                  # JavaScript reference notes
│   ├── master-roadmap/      # Development planning documents
│   ├── per-game-template/   # Template used when creating new games
│   ├── prompts/             # AI prompts and workflow prompts
│   └── workflow-documents/  # Build pipeline and workflow notes
